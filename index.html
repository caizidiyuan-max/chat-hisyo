説明・譲渡方法

1．このアプリの動き

Gmail受信
Apps Script
Gemini API
Gmail下書き

GitHubの画面を閉じても、Apps Scriptが5分ごとに確認します。最後の送信はGmailで本人が行います。

2．Gemini APIを別に設定する

APIキーはこのHTMLへ入力しません。GitHub Pagesは公開ページなので、書くと他人に盗まれる危険があります。

譲渡先の人が自分のGoogleアカウントでGemini APIキーを作り、自分のApps Scriptの「スクリプト プロパティ」へ保存します。

GEMINI_API_KEY譲渡先本人のAPIキー
DISPLAY_NAME譲渡先本人の名前
PHONE譲渡先本人の電話番号
REPLY_EMAIL譲渡先本人のGmail
普段使う有料GeminiとGemini APIは別の利用枠です。AI Studioで請求設定をしていなければ、API無料枠を使います。

3．相手へ譲渡する手順

  1. このHTMLを渡す
    GitHubへindex.htmlとして公開するか、HTMLファイルを相手へ渡します。
  2. 相手が自分のAPIキーを作る
    菜々子さんのAPIキーは渡しません。
  3. 相手が新しいApps Scriptを作る
    Apps Scriptで新しいプロジェクトを作ります。菜々子さんのプロジェクトを共同利用しません。
  4. Code.gsを貼る
    下の「Code.gsをコピー」から全部貼り付けます。
  5. 4つのプロパティを設定する
    APIキー・名前・電話番号・Gmailを相手自身の内容で入力します。
  6. setupSecretaryを実行する
    相手自身のGoogleアカウントで権限を確認し、許可します。
  7. Gmail下書きとバックアップを確認する
    自動送信されず、下書きだけ作られることを確認します。

4．容量の見方

AI Studioの使用量と上限画面で「Gemini 2.5 Flash」を確認します。通常、下書き1通で約1リクエストです。コード側は最大3通／5分なので、理論上は最大864通／日です。

5．安全チェック

  • APIキーをHTML・GitHub・チャットへ書かない
  • 相手ごとに別のApps ScriptとAPIキーを使う
  • 実メールをGitHubへ保存しない
  • 無料Gemini APIで機密メールを扱う場合はGoogleのデータ利用条件を確認する
HTMLの更新では、Gmail・Apps Script・Google Driveのバックアップは消えません。実データはHTMLとは別の場所にあります。

6．譲渡用Code.gs

秘密情報は含まれていません。APIキーは相手自身が別途設定します。

/**
 * 朝9時メール秘書：5分ごとにGmailを確認し、返信下書きだけを作ります。
 * 自動送信・削除・既読化はしません。
 */

const SECRETARY = {
  processedLabel: 'メール秘書/下書き作成済み',
  reviewLabel: 'メール秘書/要確認',
  excludedLabel: 'メール秘書/対象外',
  automaticLabel: 'メール秘書/自動通知',
  // 503が続いたため、Googleが安定版として案内しているモデルを既定にします。
  model: 'gemini-2.5-flash',
  maxThreads: 10,
  maxDraftsPerRun: 3,
  maxAiAttempts: 3,
  backupFolderName: '朝9時メール秘書バックアップ',
  backupColumns: 9
};

function setupSecretary() {
  getSettings_();
  getLabel_(SECRETARY.processedLabel);
  getLabel_(SECRETARY.reviewLabel);
  getLabel_(SECRETARY.excludedLabel);
  getLabel_(SECRETARY.automaticLabel);
  deleteSecretaryTriggers_();
  ScriptApp.newTrigger('runSecretary')
    .timeBased()
    .everyMinutes(5)
    .create();
  runSecretary();
  console.log('設定完了：5分ごとの自動確認を開始しました。自動送信はしません。');
}

function runSecretary() {
  const lock = LockService.getScriptLock();
  if (!lock.tryLock(1000)) {
    console.log('前の処理が実行中のため、今回は何もしません。');
    return;
  }

  try {
    cleanupProcessedMarkers_();

    const settings = getSettings_();
    const processedLabel = getLabel_(SECRETARY.processedLabel);
    const reviewLabel = getLabel_(SECRETARY.reviewLabel);
    const automaticLabel = getLabel_(SECRETARY.automaticLabel);
    const query = [
      'in:inbox',
      'newer_than:2d',
      '-label:"' + SECRETARY.excludedLabel + '"'
    ].join(' ');

    const threads = GmailApp.search(query, 0, SECRETARY.maxThreads);
    let created = 0;
    let skipped = 0;
    let failed = 0;

    for (let index = 0; index < threads.length; index += 1) {
      if (created >= SECRETARY.maxDraftsPerRun) break;

      const thread = threads[index];
      const messages = thread.getMessages();
      if (!messages.length) continue;

      const latest = messages[messages.length - 1];
      const messageId = latest.getId();
      if (wasProcessed_(messageId)) continue;

      const skipReason = getSkipReason_(latest, settings);
      if (skipReason) {
        markProcessed_(messageId);
        thread.addLabel(automaticLabel);
        if (!tryBackupEvent_(settings, latest, '対象外', skipReason, '')) {
          thread.addLabel(reviewLabel);
        }
        skipped += 1;
        console.log('対象外（' + skipReason + '）：' + latest.getSubject());
        continue;
      }

      try {
        const receivedBody = cleanBody_(latest.getPlainBody());
        if (!receivedBody) throw new Error('本文を読み取れませんでした');

        const reply = createReply_(latest, receivedBody, settings);
        const draftText = addSignature_(reply, settings);
        latest.createDraftReply(draftText);
        markProcessed_(messageId);
        thread.addLabel(processedLabel);
        thread.removeLabel(reviewLabel);
        if (!tryBackupEvent_(settings, latest, '下書き作成', '', draftText)) {
          thread.addLabel(reviewLabel);
        }
        created += 1;
        console.log('下書き作成：' + latest.getSubject());
      } catch (error) {
        failed += 1;
        thread.addLabel(reviewLabel);
        tryBackupEvent_(settings, latest, 'エラー', String(error), '');
        console.error('下書き作成失敗：' + latest.getSubject() + ' / ' + String(error));

        // AI全体が一時停止中なら、他のメールでも連続失敗させず次回へ回します。
        if (error && error.aiTemporary === true) {
          console.log('AIが一時的に混雑しています。残りは5分後に再試行します。');
          break;
        }
      }
    }

    console.log(
      '今回の結果：下書き' + created + '件、対象外' + skipped + '件、失敗' + failed + '件'
    );
  } finally {
    lock.releaseLock();
  }
}

function createReply_(message, receivedBody, settings) {
  const prompt = [
    'あなたは日本語のメール秘書です。受信メールへの返信下書きだけを作ってください。',
    'ルール：',
    '・丁寧で自然な敬語にする',
    '・メールに書かれていない日時、金額、約束、事実を作らない',
    '・判断できない場合は、受領したことと確認後に連絡することだけを書く',
    '・件名、宛先、署名、分析、説明文は出さない',
    '・200〜500文字程度にする',
    '',
    '差出人：' + hideEmail_(message.getFrom()),
    '件名：' + message.getSubject(),
    '本文：',
    receivedBody.slice(0, 6000)
  ].join('\n');

  const url = 'https://generativelanguage.googleapis.com/v1beta/models/'
    + encodeURIComponent(settings.model)
    + ':generateContent';

  const data = fetchAiWithRetry_(url, settings.apiKey, {
    contents: [{role: 'user', parts: [{text: prompt}]}],
    generationConfig: {temperature: 0.25, maxOutputTokens: 1000}
  });

  const parts = (((data.candidates || [])[0] || {}).content || {}).parts || [];
  const reply = parts.map(function(part) { return part.text || ''; }).join('').trim();
  if (!reply) throw new Error('AIの返信案が空でした');
  return reply.slice(0, 8000);
}

function fetchAiWithRetry_(url, apiKey, payload) {
  const waits = [0, 1500, 4000];
  let lastStatus = 0;
  let lastMessage = '';

  for (let attempt = 0; attempt < SECRETARY.maxAiAttempts; attempt += 1) {
    if (waits[attempt]) Utilities.sleep(waits[attempt]);

    try {
      const response = UrlFetchApp.fetch(url, {
        method: 'post',
        contentType: 'application/json',
        headers: {'x-goog-api-key': apiKey},
        muteHttpExceptions: true,
        payload: JSON.stringify(payload)
      });

      lastStatus = response.getResponseCode();
      const data = safeJsonParse_(response.getContentText());
      if (lastStatus >= 200 && lastStatus < 300) return data;

      lastMessage = cleanAiErrorMessage_(data);
      if (!isRetryableStatus_(lastStatus)) {
        throw createAiError_(lastStatus, lastMessage, false);
      }
    } catch (error) {
      if (error && error.aiError === true && error.aiTemporary !== true) throw error;
      lastMessage = lastMessage || String(error);
      lastStatus = lastStatus || 503;
    }
  }

  throw createAiError_(lastStatus || 503, lastMessage, true);
}

function createAiError_(status, message, temporary) {
  const suffix = message ? '（' + String(message).slice(0, 160) + '）' : '';
  const error = new Error('AI接続エラー：' + status + suffix);
  error.aiError = true;
  error.aiTemporary = temporary === true;
  return error;
}

function isRetryableStatus_(status) {
  return status === 429 || status === 500 || status === 502 || status === 503 || status === 504;
}

function safeJsonParse_(text) {
  try {
    return JSON.parse(String(text || '{}'));
  } catch (error) {
    return {};
  }
}

function cleanAiErrorMessage_(data) {
  const message = data && data.error && data.error.message ? data.error.message : '';
  return String(message).replace(/[\r\n]+/g, ' ').trim();
}

function getSettings_() {
  const p = PropertiesService.getScriptProperties();
  const settings = {
    apiKey: p.getProperty('GEMINI_API_KEY') || '',
    name: p.getProperty('DISPLAY_NAME') || '',
    phone: p.getProperty('PHONE') || '',
    email: p.getProperty('REPLY_EMAIL') || Session.getEffectiveUser().getEmail() || '',
    model: p.getProperty('GEMINI_MODEL') || SECRETARY.model,
    backupEnabled: p.getProperty('BACKUP_ENABLED') !== 'false'
  };

  const missing = [];
  if (!settings.apiKey) missing.push('GEMINI_API_KEY');
  if (!settings.name) missing.push('DISPLAY_NAME');
  if (!settings.phone) missing.push('PHONE');
  if (missing.length) throw new Error('未設定：' + missing.join(', '));
  return settings;
}

function pauseSecretary() {
  deleteSecretaryTriggers_();
  console.log('自動確認を停止しました。下書きは削除しません。');
}

function resumeSecretary() {
  getSettings_();
  deleteSecretaryTriggers_();
  ScriptApp.newTrigger('runSecretary').timeBased().everyMinutes(5).create();
  console.log('5分ごとの自動確認を再開しました。自動送信はしません。');
}

function deleteSecretaryTriggers_() {
  ScriptApp.getProjectTriggers().forEach(function(trigger) {
    if (trigger.getHandlerFunction() === 'runSecretary') {
      ScriptApp.deleteTrigger(trigger);
    }
  });
}

function getLabel_(name) {
  return GmailApp.getUserLabelByName(name) || GmailApp.createLabel(name);
}

function wasProcessed_(messageId) {
  return Boolean(PropertiesService.getScriptProperties().getProperty('DONE_' + messageId));
}

function markProcessed_(messageId) {
  PropertiesService.getScriptProperties().setProperty('DONE_' + messageId, String(Date.now()));
}

function cleanupProcessedMarkers_() {
  const properties = PropertiesService.getScriptProperties();
  const lastCleanup = Number(properties.getProperty('LAST_DONE_CLEANUP') || 0);
  const now = Date.now();
  if (now - lastCleanup < 24 * 60 * 60 * 1000) return;

  const all = properties.getProperties();
  const cutoff = now - 30 * 24 * 60 * 60 * 1000;
  const legacy = {};

  Object.keys(all).forEach(function(key) {
    if (key.indexOf('DONE_') !== 0) return;

    if (all[key] === '1') {
      // 旧版の記録は消さず、現在時刻へ変換して重複下書きを防ぎます。
      legacy[key] = String(now);
      return;
    }

    const savedAt = Number(all[key]);
    if (savedAt && savedAt < cutoff) properties.deleteProperty(key);
  });

  if (Object.keys(legacy).length) properties.setProperties(legacy, false);
  properties.setProperty('LAST_DONE_CLEANUP', String(now));
}

function getSkipReason_(message, settings) {
  if (message.isDraft()) return '下書き';
  if (isMyMessage_(message, settings.email)) return '自分から送ったメール';

  const sender = [message.getFrom(), message.getReplyTo()].join(' ');
  if (/(?:no[._-]?reply|do[._-]?not[._-]?reply|mailer-daemon|postmaster|bounce|notifications?)[@>\s]/i.test(sender)) {
    return '自動送信アドレス';
  }

  const autoSubmitted = String(message.getHeader('Auto-Submitted') || '').trim();
  if (autoSubmitted && !/^no$/i.test(autoSubmitted)) return '自動送信ヘッダー';

  const precedence = String(message.getHeader('Precedence') || '').trim();
  if (/^(?:bulk|list|junk)$/i.test(precedence)) return '一斉配信';
  if (String(message.getHeader('List-Id') || '').trim()) return 'メーリングリスト';
  if (String(message.getHeader('X-Auto-Response-Suppress') || '').trim()) return '自動通知ヘッダー';

  const subject = String(message.getSubject() || '');
  const automaticSubject = /(?:発送(?:についてのご案内|のお知らせ|中です|完了)|出荷(?:のお知らせ|完了)|配送(?:のお知らせ|状況|完了)|ご注文.*(?:受け付けました|受付|確認|完了)|注文(?:受付|確認|完了)|カード(?:利用|ご利用|ご請求)|ご請求金額|定期購入が更新|お支払い(?:完了|確認)|決済完了|領収書|認証コード|ログインコード|ワンタイムパスワード|本人確認コード|メールアドレス確認|登録完了|予約完了|shipping (?:notice|confirmation|update)|order (?:received|confirmation|shipped)|payment (?:received|confirmation)|receipt|invoice|verification code|one[- ]time passcode)/i;
  if (automaticSubject.test(subject)) return '通知メールの件名';

  return '';
}

function isMyMessage_(message, email) {
  return email && String(message.getFrom()).toLowerCase().indexOf(String(email).toLowerCase()) >= 0;
}

function cleanBody_(body) {
  return String(body || '')
    .replace(/\r\n/g, '\n')
    .split(/\n(?:On .+ wrote:|20\d{2}年.+(?:書きました|送信):)/i)[0]
    .replace(/\n{4,}/g, '\n\n')
    .trim();
}

function addSignature_(reply, settings) {
  const signature = [settings.name, 'TEL ' + settings.phone, settings.email]
    .filter(Boolean)
    .join('\n');
  return String(reply).trim() + '\n\n' + signature;
}

function hideEmail_(value) {
  return String(value || '').replace(/[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}/gi, '[メールアドレス]');
}

/**
 * バックアップは月ごとのGoogleスプレッドシートに記録します。
 * 元メールはGmailに残し、ここには処理履歴と作成した下書きを保存します。
 */
function tryBackupEvent_(settings, message, status, detail, draftText) {
  if (!settings.backupEnabled) return true;

  try {
    appendBackupEvent_(message, status, detail, draftText);
    return true;
  } catch (error) {
    console.error('バックアップ失敗：' + String(error));
    return false;
  }
}

function appendBackupEvent_(message, status, detail, draftText) {
  const spreadsheet = getMonthlyBackupSpreadsheet_(new Date());
  const sheet = spreadsheet.getSheets()[0];
  sheet.appendRow([
    new Date(),
    message.getDate(),
    status,
    String(detail || '').slice(0, 500),
    String(message.getSubject() || '').slice(0, 1000),
    String(message.getFrom() || '').slice(0, 1000),
    message.getThread().getId(),
    message.getId(),
    String(draftText || '').slice(0, 8000)
  ]);
}

function getMonthlyBackupSpreadsheet_(date) {
  const properties = PropertiesService.getScriptProperties();
  const yearMonth = Utilities.formatDate(date, 'Asia/Tokyo', 'yyyy-MM');
  const propertyName = 'BACKUP_SHEET_' + yearMonth.replace('-', '_');
  const existingId = properties.getProperty(propertyName);

  if (existingId) {
    try {
      return SpreadsheetApp.openById(existingId);
    } catch (error) {
      // 削除・移動されていた場合は、新しい月別バックアップを作ります。
      properties.deleteProperty(propertyName);
    }
  }

  const spreadsheet = SpreadsheetApp.create('メール秘書バックアップ ' + yearMonth);
  const sheet = spreadsheet.getSheets()[0];
  sheet.setName('記録');
  sheet.getRange(1, 1, 1, SECRETARY.backupColumns).setValues([[
    '処理日時', '受信日時', '状態', '詳細', '件名', '差出人',
    'スレッドID', 'メッセージID', '作成した下書き'
  ]]);
  sheet.setFrozenRows(1);

  const folder = getOrCreateBackupFolder_();
  DriveApp.getFileById(spreadsheet.getId()).moveTo(folder);
  properties.setProperty(propertyName, spreadsheet.getId());
  return spreadsheet;
}

function getOrCreateBackupFolder_() {
  const properties = PropertiesService.getScriptProperties();
  const existingId = properties.getProperty('BACKUP_FOLDER_ID');

  if (existingId) {
    try {
      return DriveApp.getFolderById(existingId);
    } catch (error) {
      properties.deleteProperty('BACKUP_FOLDER_ID');
    }
  }

  const folder = DriveApp.createFolder(SECRETARY.backupFolderName);
  properties.setProperty('BACKUP_FOLDER_ID', folder.getId());
  return folder;
}

function stopBackup() {
  PropertiesService.getScriptProperties().setProperty('BACKUP_ENABLED', 'false');
  console.log('自動バックアップを停止しました。既存のバックアップは削除しません。');
}

function startBackup() {
  PropertiesService.getScriptProperties().setProperty('BACKUP_ENABLED', 'true');
  console.log('自動バックアップを開始しました。次のメール処理からGoogle Driveへ記録します。');
}

/**
 * 初回確認用です。空の今月分バックアップ表だけをGoogle Driveへ作ります。
 * メールの送信・下書き作成・偽データの追加はしません。
 */
function prepareBackup() {
  const settings = getSettings_();
  if (!settings.backupEnabled) {
    throw new Error('バックアップが停止中です。startBackup を実行してからやり直してください。');
  }
  const spreadsheet = getMonthlyBackupSpreadsheet_(new Date());
  console.log('バックアップ準備完了：' + spreadsheet.getUrl());
}

/**

function showSevenYearBackupPlan() {
  const months = 7 * 12;
  const maxEventsPerDay = SECRETARY.maxThreads * (24 * 60 / 5);
  const maxRowsPerMonth = maxEventsPerDay * 31;
  const maxCellsPerMonth = maxRowsPerMonth * SECRETARY.backupColumns;
  console.log(
    '7年保存計画：月別' + months + '冊。最大でも1か月あたり約' +
    maxRowsPerMonth.toLocaleString() + '行（約' + maxCellsPerMonth.toLocaleString() +
    'セル）です。実際の保存量はメール件数とGoogle Driveの空き容量により変わります。'
  );
}
